203 int OutputDBManager::insertAnalisys( Analisys &a, metadata meta)
204 {
205  query = "INSERT OR REPLACE INTO " + metadata_table_name + " (JOB_TYPE, SURVEY, SITE_NAME, DATE, DATABASE, PHREEQC_VERSION, RUN_NUMBER, SAMPLE_NAME, INPUT_FILE, COORD_X, COORD_Y, COORD_Z, EPSG, TIMESTAMP, TITLE, SOLUTION) VALUES ('" +
206  a.job_type + "', '" +
207  a.survey + "', '" +
208  a.site_name + "', '" +
209  a.date + "', '" +
210  a.database + "', '" +
211  a.phreeqc_version + "', '" +
212  a.run_number + "', '" +
213  a.sample_name + "', '" +
214  a.input_file + "', '" +
215  a.coord_x + "', '" +
216  a.coord_y + "', '" +
217  a.coord_z + "', '" +
218  a.epsg + "', '" +
219  a.timestamp + "', '" +
220  meta["TITLE"] + "', '" +
221  meta["SOLUTION"] + "');";
222 
223  rc = sqlite3_exec(db, query.c_str(), 0, 0, &err_message);
224 
225  if (rc == SQLITE_OK)
226  {
227  this->queryResult(rc, "inserting analisys");
228  a.id = sqlite3_last_insert_rowid(db);
229  this->a_id = a.id;
230  }
231  else
232  {
233  std::cerr << "\033[1;31mSQL ERROR " << rc << ": " << err_message << "\033[0m" << std::endl;
234 
235  switch (rc)
236  {
237  case SQLITE_CONSTRAINT:
238  std::cerr << "\033[1;31mAnalysis " << a.input_file << " may already be in the database computed with the same PHREEQC database and it is considered as a duplicate IGNORED.\033[0m" << std::endl;
239  break;
240  default:
241  std::cerr << "" << std::endl;
242  }
243  }
244 
245  return rc;
246 }

354 bool OutputDBManager::selectValuesfromAnalisys( Analisys &a, int analysis_id)
355 {
356  char **res;
357  int row, column;
358  SolutionComposition sc;
359 
360  query = "SELECT * FROM " + metadata_table_name + " WHERE ID='" + to_string(analysis_id) + "';";
361  rc = sqlite3_get_table(db, query.c_str(), &res, &row, &column, &err_message);
362  this->queryResult(rc, "selecting analisys");
363 
364  if (row == 0 || column == 0)
365  {
366  std::cerr << __FUNCTION__ << " ERROR : Unexisting analysis ID " << analysis_id << std::endl;
367  return false;
368  }
369 
370  a.id = atoi(res[0 + column]);
371  a.job_type = res[1 + column];
372  a.survey = res[2 + column];
373  a.site_name = res[3 + column];
374  a.date = res[4 + column];
375  a.database = res[5 + column];
376  a.phreeqc_version = res[6 + column];
377  a.run_number = res[7 + column];
378  a.sample_name = res[8 + column];
379  a.input_file = res[9 + column];
380  a.coord_x = res[10 + column];
381  a.coord_y = res[11 + column];
382  a.coord_z = res[12 + column];
383  a.epsg = res[13 + column];
384  a.timestamp = res[14 + column];
385 
386  sqlite3_free_table(res);
387 
388  return true;
389 }

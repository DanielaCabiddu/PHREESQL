78inline
79void InputDBManager::selectValuesFromMetadata(metadata &meta, int ANALYSIS_num)
80{
81 char **res;
82 int row, column;
83
84 meta.clear();
85
86 query = "SELECT * FROM " + metadata_table_name + " WHERE ID='" + to_string(ANALYSIS_num) + "';";
87 rc = sqlite3_get_table(db, query.c_str(), &res, &row, &column, &err_message);
88 this->queryResult(rc, "selecting data from metadata");
89
90 for (int i = 0; i < column; i++)
91 {
92 meta.insert(pair<string, string>(res[i], res[i + column]));
93 }
94 sqlite3_free_table(res);
95}

97inline
98void InputDBManager::selectValuesFromSolution(vector<SolutionInput> &inputs, int ANALYSIS_num)
99{
100 char **res;
101 int row, column;
102 SolutionInput input;
103
104 query = "SELECT * FROM SOLUTION_INPUT WHERE ANALYSIS_ID='" + to_string(ANALYSIS_num) + "';";
105 rc = sqlite3_get_table(db, query.c_str(), &res, &row, &column, &err_message);
106 this->queryResult(rc, "selecting data from solution_input");
107
108 for (int i = 5; i < (row + 1) * column; i = i + 5)
109 {
110 input.analisys_id = atoi(res[i + 1]);
111 input.solution_number = res[i + 2];
112 input.unit = res[i + 3];
113 input.ppm = res[i + 4];
114 inputs.push_back(input);
115 }
116 sqlite3_free_table(res);
117}

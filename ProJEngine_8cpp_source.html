36inline
37void phreesqlib::ProjEngine::epsg2epsg (const double epsg1_x,
38 const double epsg1_y,
39 const double epsg1_z,
40 const unsigned int epsg1,
41 const unsigned int epsg2,
42 double &epsg2_x,
43 double &epsg2_y,
44 double &epsg2_z )
45{
46
47 PJ_CONTEXT *c = proj_context_create();
48 PJ *transformation = proj_create_crs_to_crs(c,
49 std::string("EPSG:" + std::to_string(epsg1)).c_str(),
50 std::string("EPSG:" + std::to_string(epsg2)).c_str(),
51 nullptr);
52
53 epsg2epsg(transformation, epsg1_x, epsg1_y, epsg1_z, epsg2_x, epsg2_y, epsg2_z);
54
55 proj_context_destroy (c);
56}

58inline
59void phreesqlib::ProjEngine::epsg2epsg (PJ *transformation,
60 const double epsg1_x,
61 const double epsg1_y,
62 const double epsg1_z,
63 double &epsg2_x,
64 double &epsg2_y,
65 double &epsg2_z )
66{
67
68 if (!transformation)
69 {
70 std::cerr << "ERROR PROJ" << std::endl;
71
73 exit(1);
74 }
75
76 PJ_COORD coords;
77 coords.xyz.x = epsg1_x;
78 coords.xyz.y = epsg1_y;
79 coords.xyz.z = epsg1_z;
80
81// std::cout << "DEGREE : " << coords.lpz.phi << " - " << coords.lpz.lam << " - " << coords.lpz.z << std::endl;
82
83 PJ_COORD converted_coords = proj_trans(transformation, PJ_DIRECTION::PJ_FWD, coords);
84
85// std::cout << "CONVERTED : " << converted_coords.xyz/.x << " - " << converted_coords.xyz.y << " - " << converted_coords.xyz.z << std::endl;
86
87 epsg2_x = converted_coords.xyz.x;
88 epsg2_y = converted_coords.xyz.y;
89 epsg2_z = converted_coords.xyz.z;
90}

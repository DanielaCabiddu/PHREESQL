94inline
95vector<vector<string> > DataManager::getData(string table_name, string value, string column_name, string timestamp)
96{
97 char **res;
98 int row, column;
99
100 query = "SELECT " + toUpperString(column_name) + ", TIMESTAMP FROM " + toUpperString(table_name) + " JOIN ANALISYS ON ANALISYS_ID LIKE ANALISYS.ID WHERE " + "ANALISYS.TIMESTAMP>=\"" + timestamp + "\" AND " + valueColName(table_name) + "=\"" + value + "\" ORDER BY TIMESTAMP;";
101 rc = sqlite3_get_table(db, query.c_str(), &res, &row, &column, &err_message);
102 this->queryResult(rc, "getData");
103
104 vector<vector<string> > data;
105 vector<string> elems;
106 for (int i = 2; i < (row + 1) * column; i = i + 2)
107 {
108 elems.clear();
109 elems.push_back(res[i]);
110 elems.push_back(res[i + 1]);
111 data.push_back(elems);
112 };
113
114 sqlite3_free_table(res);
115 return data;
116}

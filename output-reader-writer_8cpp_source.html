34const string DoubleQuotes(const string value)
35{
36 string retval;
37 for (auto ch : value)
38 {
39 if (ch == '\'')
40 {
41 retval.push_back('\'');
42 }
43 retval.push_back(ch);
44 }
45 return retval;
46}

60 ifstream input_file;
61 string line, key, value;
62 input_file.open(path, ios::in); //open a file to perform read operation using file object
63

149inline
150bool OutputReaderWriter::readFile(string path)
151{
152 ifstream input_file;
153 int i = 0;
154
155 input_file.open(path, ios::in); //open a file to perform read operation using file object
156 if (input_file.is_open())
157 { //checking whether the file is open
158 string line;
159
160 while (getline(input_file, line))
161 { //read data from file object and put it into string.
162 i++;
163 if (line.rfind('-', 0) == 0)
164 {
165 if (strstr(line.c_str(), "Solution composition") != NULL)
166 {
167 readSolutionComposition(input_file, i);
168 }
169 else if (strstr(line.c_str(), "Description of solution") != NULL)
170 {
171 readDescriptionOfSolution(input_file, i);
172 }
173 else if (strstr(line.c_str(), "Distribution of species") != NULL)
174 {
175 readDistributionOfSpecies(input_file, i);
176 }
177 else if (strstr(line.c_str(), "Distribution of alkalinity") != NULL)
178 {
179 readDistributionOfAlkalinity(input_file, i);
180 }
181 else if (strstr(line.c_str(), "Saturation indices") != NULL)
182 {
183 readSaturationIndices(input_file, i);
184 }
185 }
186 }
187 input_file.close(); //close the file object.
188
189 checkSection(sc_list.size(), "Solution composition", path);
190 checkSection(des_list.size(), "Description of solution", path);
191 checkSection(dis_list.size(), "Distribution of species", path);
192 checkSection(dis_list.size(), "Distribution of alkalinity", path);
193 checkSection(si_list.size(), "saturation indices", path);
194
195 return true;
196 }
197 return false;
198}

229inline
230void OutputReaderWriter::readDescriptionOfSolution(ifstream &file, int &row)
231{
232 string line, key, value;
233
234 removeHeaders1(file, "", row);
235
236 while (getline(file, line) && line.size() > 1)
237 {
238 row++;
239 istringstream iss(line);
240 DescriptionOfSolution des;
241 key = trim(line.substr(0, line.find('=')));
242 value = trim(line.substr(line.find('=') + 1, line.size()));
243
244 // Get the first occurrence
245 std::string toSearch = std::to_string(char(176));
246 size_t pos = key.find(toSearch);
247
248 if (pos < key.length())
249 {
250 std::string replaceStr = key;
251 // Repeat till end is reached
252 while( pos != std::string::npos)
253 {
254 // Replace this occurrence of Sub String
255 key.replace(pos, toSearch.size(), "U+2103");
256 // Get the next occurrence from the current position
257 pos =key.find(toSearch, pos + replaceStr.size());
258 }
259
260 key = replaceStr;
261 }
262
263 des.key_name = key;
264 des.value = value;
265 this->des_list.push_back(des);
266 }
267 if (line.size() == 0)
268 {
269 row++;
270 }
271 //printDescriptionOfSolution(this->des_list);
272}

276{
277 string line, value;
278
279 removeHeaders1(file, "Species", row);
280
281 while (getline(file, line) && line.size() > 1)
282 {
283 row++;
284 istringstream iss(line);
285 DistributionOfSpecies dis;
286
287 for (int i = 0; i < 6; i++)
288 {
289 if (iss >> value)
290 {
291 switch (i)
292 {
293 case 0:
294 dis.species = value;
295 break;
296 case 1:
297 dis.molality = value;
298 break;
299 case 2:
300 dis.activity = value;
301 break;
302 case 3:
303 dis.log_molality = value;
304 break;
305 case 4:
306 dis.log_activity = value;
307 break;
308 case 5:
309 dis.log_gamma = value;
310 break;
311 }
312 }
313 }
314 this->dis_list.push_back(dis);
315 }
316 if (line.size() == 0)
317 {
318 row++;

135 inline
136 bool PhreeqcInterface::readIOMFiles(string ifile_path, string ofile_path, string meta_path)
137 {
138  if (i_file.readFile(ifile_path) && o_file.readFile(ofile_path) && o_file.readMetadata(meta_path))
139  {
140  this->createInputTables();
141  this->createOutputTables();
142  o_manager->createAnalisysTable();
143 
144  int result = o_manager->insertAnalisys(o_file.a, i_file.meta);
145 
146  if (result == SQLITE_OK)
147  {
148  i_manager->insertSolutionInputs(i_file.input_list, (o_file.a).id);
149 
150  o_manager->insertSolutionComposition(o_file.sc_list);
151  o_manager->insertDescriptionOfSolution(o_file.des_list);
152  o_manager->insertDistributionOfSpecies(o_file.dis_list);
153  o_manager->insertDistributionOfAlkalinity(o_file.alk_list);
154  o_manager->insertSaturationIndices(o_file.si_list);
155  }
156 
157  return true;
158  }
159  else
160  {
161  cerr << "Error in opening either input/output or metadata file " << endl;
162  return false;
163  }

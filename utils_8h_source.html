100 inline bool mkpath (const std::string dir)
101 {
102  char tmp[256];
103  char *p = NULL;
104  size_t len;
105 
106  snprintf(tmp, sizeof(tmp),"%s",dir.c_str());
107  len = strlen(tmp);
108 
109  int error = 0;
110 
111  if(tmp[len - 1] == separator().c_str()[0])
112  tmp[len - 1] = 0;
113 
114  for(p = tmp + 1; *p; p++)
115  {
116  if(*p == separator().c_str()[0])
117  {
118  *p = 0;
119 
120  if (!dirExists(tmp))
121  {
122 #ifdef WIN32
123  error = _mkdir(tmp);
124 #else
125  error = mkdir(tmp, S_IRWXU);
126 #endif
127 
128 
129  if (error != 0)
130  {
131  std::cerr << "error creating folder " << tmp << std::endl;
132  return false;
133  }
134  }
135  *p = separator().c_str()[0];
136  }
137  }
138 
139 #ifdef WIN32
140  error = _mkdir(tmp);
141 #else
142  error = mkdir(tmp, S_IRWXU);
143 #endif
144 
145  if (error != 0)
146  {
147  std::cerr << "error creating folder " << tmp << std::endl;
148  return false;
149  }
150 
151  return true;
152 }

154 inline std::string ltrim(const std::string &s)
155 {
156  size_t start = s.find_first_not_of(WHITESPACE);
157  return (start == std::string::npos) ? "" : s.substr(start);
158 }
159 
160 inline std::string rtrim(const std::string &s)
161 {
162  size_t end = s.find_last_not_of(WHITESPACE);
163  return (end == std::string::npos) ? "" : s.substr(0, end + 1);
164 }

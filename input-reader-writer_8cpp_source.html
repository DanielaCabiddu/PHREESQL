32inline
33bool InputReaderWriter::readFile(string path)
34{
35 string solution;
36 fstream input_file;
37
38 input_file.open(path, ios::in); //open a file to perform read operation using file object
39
40 if (input_file.is_open())
41 { //checking whether the file is open
42 string line;
43
44 while (getline(input_file, line))
45 { //read data from file object and put it into string.
46 istringstream iss(line);
47 string key, value;
48 if (!(iss >> key >> value))
49 {
50 if (key.compare("END") == 0)
51 {
52 break;
53 }
54 }
55
56 if (key.rfind("#", 0) == 0)
57 {
58 continue;
59 }
60
61 if (key.compare("DATABASE") == 0)
62 {
63 meta.insert(pair<string, string>(key, line.substr(line.find(key) + key.size(), line.size())));
64 }
65 else if (key.compare("TITLE") == 0)
66 {
67 meta.insert(pair<string, string>(key, line.substr(line.find(key) + key.size(), line.size())));
68 }
69 else if (key.compare("SOLUTION") == 0)
70 {
71 meta.insert(pair<string, string>(key, value));
72 solution = value;
73 }
74 else
75 {
76// if (key.compare("units") != 0)
77// {
78 SolutionInput input;
79 input.unit = trim(key);
80 input.ppm = trim(line.substr(line.find(key) + key.size(), line.size()));
81 input.solution_number = solution;
82 input_list.push_back(input);
83// }
84 }
85 }
86 input_file.close(); //close the file object.
87
88 if (meta.find("TITLE") == meta.end())
89 meta.insert(pair<string, string>("TITLE", ""));
90
91 if (meta.find("SOLUTION") == meta.end())
92 meta.insert(pair<string, string>("SOLUTION", ""));
93
94 return true;
95 }
96 return false;
97}

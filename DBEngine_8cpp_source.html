48 void phreesqlib::DBEngine::convert_epsg (const int epsg, const std::vector<EPSG_CONVERT_TYPE> types, std::vector<std::string> outputs)
49 {
50  PhreeqcInterface *matrac_reader = new PhreeqcInterface(db_filename);
51  std::vector<std::vector<std::pair<std::string, std::string>>> metadata_table = matrac_reader->getMetadata ();
52 
53  std::vector<std::string> ids;
54  std::vector<double> x_vect;
55  std::vector<double> y_vect;
56  std::vector<double> z_vect;
57 
58  phreesqlib::ProjEngine proj_engine;
59 
60  for (unsigned int i=0; i < metadata_table.size(); i++)
61  {
62  double x = DBL_MAX;
63  double y = DBL_MAX;
64  double z = DBL_MAX;
65  unsigned int epsg_curr = UINT_MAX;
66  std::string id;
67 
68  for (unsigned int j=0; j < metadata_table.at(i).size(); j++)
69  {
70  try {
71 
72  if (metadata_table.at(i).at(j).first.compare("EPSG") == 0)
73  epsg_curr = std::atoi(metadata_table.at(i).at(j).second.c_str());
74  else if (metadata_table.at(i).at(j).first.compare("ID") == 0)
75  id = metadata_table.at(i).at(j).second.c_str();
76  else if (metadata_table.at(i).at(j).first.compare("COORD_X") == 0)
77  x = std::stod(metadata_table.at(i).at(j).second.c_str());
78  else if (metadata_table.at(i).at(j).first.compare("COORD_Y") == 0)
79  y = std::stod(metadata_table.at(i).at(j).second.c_str());
80  else if (metadata_table.at(i).at(j).first.compare("COORD_Z") == 0)
81  {
82  try
83  {
84  z = std::stod(metadata_table.at(i).at(j).second.c_str());
85  } catch (exception e)
86  {
87  std::cerr << "\033[1;33m Warning: [" << i << "] Z coordinate not available. Set to 0.0. \033[0m" << std::endl;
88  z=0.0;
89  }
90  }
91 
92  } catch (exception e) {
93  std::cerr << metadata_table.at(i).at(j).second.c_str() << std::endl;
94  std::cerr << e.what() << std::endl;exit(1);
95  }
96  }
97 
98  double x_converted = DBL_MAX;
99  double y_converted = DBL_MAX;
100  double z_converted = DBL_MAX;
101 
102  if (epsg == epsg_curr)
103  {
104  x_converted = x;
105  y_converted = y;
106  z_converted = z;
107  }
108  else
109  proj_engine.epsg2epsg(x, y, z, epsg_curr, epsg, x_converted, y_converted, z_converted);
110 
111  x_vect.push_back(x_converted);
112  y_vect.push_back(y_converted);
113  z_vect.push_back(z_converted);
114 
115  ids.push_back(id);
116  }
117 
118  for (unsigned int t=0; t < types.size(); t++)
119  {
120  if (types.at(t) == TABLE)
121  {
122  bool success = matrac_reader->create_and_insert_EpsgTable(outputs.at(t), ids, x_vect, y_vect, z_vect);
123 
124  if (success)
125  std::cout << "Created table " << outputs.at(t) << " in DB : " << db_filename << std::endl;
126  else
127  std::cerr << "\033[1;31mERROR in creating table " << outputs.at(t) << " in DB : " << db_filename << "\033[0m" << std::endl;
128  }
129 
130  if (types.at(t) == DB)
131  {
132  matrac_reader->copyDB(outputs.at(t));
133 
134  PhreeqcInterface *matrac_reader_copy = new PhreeqcInterface(outputs.at(t));
135 
136  bool success = matrac_reader_copy->updateAnalysisEPSG (epsg, ids, x_vect, y_vect);
137 
138  delete matrac_reader_copy;
139 
140  if (success)
141  std::cout << "Created DB : " << outputs.at(t) << std::endl;
142  else
143  std::cerr << "\033[1;31mERROR in creating DB : " << outputs.at(t) << "\033[0m" << std::endl;
144  }
145 
146  if (types.at(t) == CSV)
147  {
148  std::string filename = outputs.at(t);
149 
150  ofstream out_file;
151  out_file.open(filename.c_str());
152 
153  for (unsigned int i=0; i < metadata_table.at(0).size()-1; i++)
154  out_file << metadata_table.at(0).at(i).first << ";";
155 
156  out_file << metadata_table.at(0).at(metadata_table.at(0).size()-1).first << std::endl;
157 
158  for (unsigned int i=0; i < metadata_table.size(); i++)
159  {
160  for (unsigned int j=0; j < metadata_table.at(i).size(); j++)
161  {
162  if (metadata_table.at(i).at(j).first.compare("EPSG") == 0)
163  out_file << epsg << ";";
164  else if (metadata_table.at(i).at(j).first.compare("COORD_X") == 0)
165  out_file << std::fixed << x_vect.at(i) << ";";
166  else if (metadata_table.at(i).at(j).first.compare("COORD_Y") == 0)
167  out_file << std::fixed << y_vect.at(i) << ";";
168  else if (metadata_table.at(i).at(j).first.compare("COORD_Z") == 0)
169  out_file << std::fixed << z_vect.at(i) << ";";
170  else
171  {
172  out_file << metadata_table.at(i).at(j).second ;
173 
174  if (j == metadata_table.at(i).size()-1)
175  out_file << std::endl;
176  else
177  out_file << ";";
178  }
179  }
180  }
181 
182  out_file.close();
183 
184  std::cout << "Saved CSV : " << filename << std::endl;
185  }
186  }
187 
188  delete matrac_reader;
189 }
